create database cantina;

use cantina;

create table cliente (cpf bigint unsigned primary key not null, rg varchar(14), nome varchar(200) not null, email varchar(200), tel bigint, dataNasc datetime, endereco varchar(300), prevPagamento datetime );

create table usuario(login varchar(15) primary key, senha varchar(15), nome varchar(50), telefone bigint, email varchar(40), adm bool);

create table compra (id int unsigned primary key not null auto_increment, dataCompra datetime, total double, dataPagamento datetime, fk_cliente int unsigned not null  references cliente on delete set null, fk_usuario varchar(15) not null references usuario on delete set null, key (fk_cliente), key (fk_usuario));

create table produto(id int unsigned primary key not null auto_increment, descricao varchar(50) unique, precoUnitario double);

create table itemProduto(id int unsigned primary key not null auto_increment, quantidade int unsigned, fk_compra int unsigned not null references compra on delete set null, fk_produto int unsigned not null references produto on delete set null, key(fk_compra), key(fk_produto));

create table fornecedor(cnpj bigint unsigned primary key not null, tel varchar(15), email varchar(150), nome varchar(200), endereco varchar(255));

create table notaFiscal(id int unsigned primary key not null auto_increment, nBoleto bigint unsigned, precoTotal double, dataPagamento datetime, dataVencimento datetime, dataCompra datetime, fk_fornecedor bigint unsigned not null references fornecedor, key (fk_fornecedor));

create table insumo(id int unsigned primary key not null auto_increment, descricao varchar(50), precoUnitario double);

create table itemInsumo(id int unsigned primary key not null auto_increment, quantidade int, fk_insumo int unsigned not null references insumo, fk_notafiscal int not null references notaFiscal, key (fk_insumo), key (fk_notafiscal));
